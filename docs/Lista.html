def each
	actual = @tail
	for i in 0 .. (@size - 1)
		yield actual.value
		actual = actual.next
	end
end

def insert(var)
	if var.instance_of?(Array)
		var.each do |i|
			ins(i)
		end
	else
		ins(var)
	end
end

def pop
	if @tail.nil? == false
		nuevo = @tail.next
		@tail = nuevo
		@size -= 1
	end
end

def shift

		@size -= 1
